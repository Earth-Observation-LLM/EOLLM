\begin{frame}{The Task}

\begin{columns}[T]
\begin{column}{0.55\textwidth}
    \textbf{Given:} Satellite + street-level imagery of a location\\[0.5em]
    \textbf{Ask:} Multiple-choice questions about urban properties\\[0.5em]
    \textbf{Answer:} A single letter (A, B, C, or D)

    \vspace{1em}

    \begin{tabular}{@{}ll@{}}
    \toprule
    \textbf{Scale} & \textbf{Count} \\
    \midrule
    Cities & 40 worldwide \\
    Topics & 14 urban categories \\
    Train samples & 26,931 \\
    Validation samples & 6,984 \\
    \bottomrule
    \end{tabular}
\end{column}
\begin{column}{0.42\textwidth}
    \small
    \textbf{Topics covered:}\\[0.3em]
    Land use, building height, road type, road surface, urban density, green space, junction type, amenity richness, transit density, camera direction, mismatch detection (4 variants)
\end{column}
\end{columns}

\end{frame}

\begin{frame}{Setup}

\begin{columns}[T]
\begin{column}{0.48\textwidth}
    \textbf{Model}\\[0.3em]
    \begin{tabular}{@{}ll@{}}
    Base & Qwen3.5-4B \\
    Method & bf16 LoRA \\
    Trainable & 38.8M / 4.58B \\
    & \textbf{(0.85\% of params)} \\
    \end{tabular}

    \vspace{1em}

    \textbf{Hardware}\\[0.3em]
    \begin{tabular}{@{}ll@{}}
    GPU & RTX 5090 (32 GB) \\
    Peak VRAM & 17 GB (52\%) \\
    Wall clock & 96 minutes \\
    \end{tabular}
\end{column}
\begin{column}{0.48\textwidth}
    \textbf{Why these choices?}\\[0.3em]
    \small
    \begin{tabular}{@{}lp{3.8cm}@{}}
    \toprule
    \textbf{Choice} & \textbf{Reason} \\
    \midrule
    bf16, not 4-bit & Qwen3.5 degrades with quantization \\[0.2em]
    LoRA r=16 & Balanced capacity for 27k samples \\[0.2em]
    All-linear & Vision + language both need adaptation \\[0.2em]
    LR 2e-4 & Standard LoRA starting point \\
    \bottomrule
    \end{tabular}
\end{column}
\end{columns}

\end{frame}

% ============================================================
\begin{frame}{Hyperparameters}

\centering
\begin{tabular}{@{}lll@{}}
\toprule
\textbf{Parameter} & \textbf{Value} & \textbf{Rationale} \\
\midrule
Batch size & $4 \times 6 = 24$ effective & Largest before OOM \\
Learning rate & $2 \times 10^{-4} \to 0$ & Linear decay with 40-step warmup \\
LoRA rank & 16 (alpha = 16) & Balanced for 27k samples \\
max\_seq\_length & 8192 tokens & p99 = 1,535; generous headroom \\
Image max edge & 768 px & Profile-matched to RTX 5090 \\
Optimizer & AdamW 8-bit & VRAM efficient \\
Training steps & 400 & $\sim$35\% of 1 epoch \\
\bottomrule
\end{tabular}

\vspace{1em}
\small
Token budget measured on 300 samples using the actual Qwen3.5 processor.\\
Batch size determined by forward+backward probe at bs = 1, 2, 4, 8.

\end{frame}
